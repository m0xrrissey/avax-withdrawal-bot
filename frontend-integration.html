  return `https://t.me/${BOT_USERNAME}?start=addr_${walletAddress}`;
}

// Open Telegram with pre-filled wallet address
function openTelegramBot(walletAddress) {
  const telegramLink = generateTelegramLink(walletAddress);
  window.open(telegramLink, '_blank');
}
